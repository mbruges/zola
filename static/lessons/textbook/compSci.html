What is Hexidecimal?


def text_to_hex(text):
    hex_codes = []
    for char in text:
        hex_code = hex(ord(char))[2:]  # Remove the '0x' prefix
        hex_codes.append(hex_code)
    return ' '.join(hex_codes)
